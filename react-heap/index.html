🚮 Trevor’s Tip #1

“Use it or lose it”

🐢 Before

const Sidebar = ({ ungated }) => {
  const oldFeature = <OldFeature />;
  const newFeature = <NewFeature />;
  return ungated
    ? newFeature
    : oldFeature;
};

🐇 After

const Sidebar = ({ ungated }) => {
  return ungated
    ? <NewFeature />
    : <OldFeature />;
};

♻️ Trevor’s Tip #2:

“Never refuse to reuse”

🐢 Before

const UpgradeNudge = () =>
  <UIHelpLink
    content={{
      header: <H4>Give us more 💰</H4>,
      body: <p>💸 Make 👏🏻 it 👏🏻 rain! 💸</p>,
    }}
  >
    A friendly reminder 🙃
  </UIHelpLink>

🤥 After?

class UpgradeNudge extends PureComponent {
  ...
}

👯‍♀️ Same JSX, same result

{
  $$typeof: Symbol(react.element),
  props: ...,
  type: UIHelpLink,
}

🐇 After (for real!)

const upgradeNudge =
  <UIHelpLink
    content={{
      header: <H4>Give us more 💰</H4>,
      body: <p>💸 Make 👏🏻 it 👏🏻 rain! 💸</p>,
    }}
  >
    A friendly reminder 🙃
  </UIHelpLink>

🏦 Trevor’s Tip #3:

“Save it for later”

🐢 Before

const ActionsDropdown = () =>
  <UIDropdown buttonText="Actions">
    <UIList>
      <UIButton>✏️ Edit</UIButton>
      <UIButton>💾 Save</UIButton>
      <UIButton>💥 Destroy</UIButton>
    </UIList>
  </UIDropdown>

🐇 After (1 of 2)

const ActionsDropdown = () =>
  <UIDropdown
    buttonText="Actions"
    Content={DropdownContent}
  />

🐇 After (2 of 2)

const DropdownContent = () =>
  <UIList>
    <UIButton>✏️ Edit</UIButton>
    <UIButton>💾 Save</UIButton>
    <UIButton>💥 Destroy</UIButton>
  </UIList>
